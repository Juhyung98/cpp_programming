•
int main(void)

{

   int i1 = 10, i2 = 30;

   double d1 = 0.5, d2 = 3.5;

   string s1 = "Hello", s2 = "World";

 

   cout << add (i1, i2) << ", "

        << add (d1, d2) << ", "

}
